popt, pcov = curve_fit(exp, esp, nc,sigma=delta_nc,absolute_sigma=True,  p0=[40000,-0.1])
I0a,mu=popt

delta_I0=nc[0]**0.5

incertezas=np.sqrt(np.diag(pcov))
delta_I0a, delta_mu = incertezas 

esp_ajuste = np.linspace(min(esp), max(esp), 100)
