plt.figure(figsize=(16, 9))
for Cp in Cp_list:
    # Inicializao para cada Cp
    u_dimless = np.zeros(Ny)

    # Loop temporal
    for _ in range(n_steps):
        u_old = u_dimless.copy()
        for j in range(1, Ny - 1):
            u_dimless[j] = u_old[j] + dt_dimless * Cp + \
                           alpha * (u_old[j+1] - 2*u_old[j] + u_old[j-1])
        u_dimless[0] = 0.0
        u_dimless[-1] = 0.0

    # Soluo analtica
    u_steady_analytical = (Cp / 2.0) * y_dimless * (1.0 - y_dimless)

    # Plot do perfil final para este Cp
    plt.plot(y_dimless, u_dimless, label=f'Numrico $C_p={Cp}$')
    plt.plot(y_dimless, u_steady_analytical, '--', label=f'Analtico $C_p={Cp}$')

# Labels e grid
