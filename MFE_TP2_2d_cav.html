def calcular_velocidades(psi, dl):
    u = np.zeros_like(psi)
    v = np.zeros_like(psi)
    u[1:-1, 1:-1] = -(psi[1:-1, 2:] - psi[1:-1, 0:-2]) / (2 * dl)
    v[1:-1, 1:-1] =  (psi[2:, 1:-1] - psi[0:-2, 1:-1]) / (2 * dl)
    return u, v

def aplicar_condicoes_fronteira_omega(omega, psi, dl, Nl):
    # Parede superior (parede 1)
    omega[:, Nl - 1] = 2 * psi[:, Nl - 2] / dl**2 - 2 / dl
    # Parede inferior (parede 3)
    omega[:, 0] = 2 * psi[:, 1] / dl**2
    # Parede esquerda (parede 4)
    omega[0, :] = 2 * psi[1, :] / dl**2
    # Parede direita (parede 2)
    omega[Nl - 1, :] = 2 * psi[Nl - 2, :] / dl**2
    return omega

def atualizar_vorticidade(omega, u, v, dl, dt, Re):
    omega_new = omega.copy()
    convection = (
        u[1:-1,1:-1] * (omega[2:,1:-1] - omega[0:-2,1:-1]) / (2 * dl) +
        v[1:-1,1:-1] * (omega[1:-1,2:] - omega[1:-1,0:-2]) / (2 * dl)
    )
    diffusion = (
        (omega[2:,1:-1] - 2 * omega[1:-1,1:-1] + omega[0:-2,1:-1]) +
        (omega[1:-1,2:] - 2 * omega[1:-1,1:-1] + omega[1:-1,0:-2])
    ) / dl**2

    omega_new[1:-1,1:-1] = omega[1:-1,1:-1] + dt * (-convection + (1 / Re) * diffusion)
    return omega_new

def resolver_poisson_psi(psi, omega, dl, dt_prime, Nm):
    for _ in range(Nm):
        psi[1:-1, 1:-1] += dt_prime * (
            (psi[2:,1:-1] - 2 * psi[1:-1,1:-1] + psi[0:-2,1:-1]) / dl**2 +
            (psi[1:-1,2:] - 2 * psi[1:-1,1:-1] + psi[1:-1,0:-2]) / dl**2 -
            omega[1:-1,1:-1]
        )
    return psi

# Loop principal
for t in range(Nt):
    u_vel, v_vel = calcular_velocidades(psi, dl)
    omega = aplicar_condicoes_fronteira_omega(omega, psi, dl, Nl)
    omega = atualizar_vorticidade(omega, u_vel, v_vel, dl, dt, Re)
    psi = resolver_poisson_psi(psi, omega, dl, dt_prime, Nm)

    if t % 500 == 0:
        print(f"Passo de tempo {t}/{Nt}")
