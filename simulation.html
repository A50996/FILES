Calculo das N iteraes:

$x(n+1)=x(n)+v_x(n) \Delta t+\frac{1}{2} a_x(n)(\Delta t)^{2}$

$y(n+1)=y(n)+v_y(n) \Delta t+\frac{1}{2} a_y(n)(\Delta t)^{2}$

$-------------$

$\overrightarrow{F}_{y_i}(n+1) = k_0 \frac{q \cdot q_i \cdot [y(n+1) - y_i ]}{\left(\sqrt{[x_i - x(n+1)]^2 + [y_i - y(n+1)]^2}\right)^3}$

$\overrightarrow{F}_{y}(n+1) = \sum_{i=1}^4 \overrightarrow{F}_{y_i}(n+1)$

$\overrightarrow{a}_y(n+1) = \overrightarrow{F}_{y}(n+1)$

$-------------$

$\overrightarrow{F}_{x_i}(n+1) = k_0 \frac{q \cdot q_i \cdot [x(n+1)-x_i]}{\left(\sqrt{[x_i - x(n+1)]^2 + [y_i - y(n+1)]^2}\right)^3}$

$\overrightarrow{F}_{x}(n+1) = \sum_{i=1}^4 \overrightarrow{F}_{x_i}(n+1)$

$\overrightarrow{a}_x(n+1) = \overrightarrow{F}_{x}(n+1)$

$-------------$

$ v_x(n+1)=v_x(n)+\frac{1}{2}\left[a_x(n)+a_x(n+1)\right] \Delta t $

$ v_y(n+1)=v_y(n)+\frac{1}{2}\left[a_y(n)+a_y(n+1)\right] \Delta t$

#  Dado que para ns  mais fcil
#  pensar assim, fizemos as seguintes 
#  alteraes nas equaes:
#  n->n-1 e n+1->n
N=1000
Dt=0.01
for n in range(1,N): 
   


 xn=x[n-1]+v_x[n-1]*Dt+0.5*a_x[n-1]*Dt**2 #calculo de x(n)
 yn=y[n-1]+v_y[n-1]*Dt+0.5*a_y[n-1]*Dt**2 #calculo de y(n)
 
 y=np.append(y,yn) #juntar s listas de valores  
 x=np.append(x,xn)


 f_x_vetor=k_0*(q*q_fixas)*(x[n]-x_qfixas)/((x_qfixas-x[n])**2+(y_qfixas-y[n])**2)**1.5 
 #calculo de todos os F_xi, retorna um array com os 4

 f_x=sum(f_x_vetor)#soma todos os elementos do array
 a_x=np.append(a_x,f_x)#fx=ax, guarda na lista de ax

 #repete mas para as componentes y
 f_y_vetor=k_0*(q*q_fixas)*(y[n]-y_qfixas)/((y_qfixas-y[n])**2+(x_qfixas-x[n])**2)**1.5
 f_y=sum(f_y_vetor)
 a_y=np.append(a_y,f_y)

 v_xn=v_x[n-1]+0.5*(a_x[n-1]+a_x[n])*Dt#calculo de vx(n)/vy(n)
 v_yn=v_y[n-1]+0.5*(a_y[n-1]+a_y[n])*Dt

 v_y=np.append(v_y,v_yn)#juntar s listas de valores
 v_x=np.append(v_x,v_xn)
