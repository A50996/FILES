m = 1.0      # Massa (kg)
l = 1.0      # Comprimento (m)
g = 9.81     # Gravidade (m/s)
b = np.array([0.0 , 1.0 , 0.5])     # Coeficiente de atrito
H = np.array([2.0 , 2.0 , 4.0])      # Fora de propulso constante (N)

# Adimensionalizar parmetros
b_mgl = b / (m * np.sqrt(g * l))
H_mg = H / (m * g)


A=[0.5 , 0.5, 0.3 , 0.1]
B=[0.2 , 0.8 , 0.1 , 0]

Adimencionamento

$\tau=t\sqrt{\frac{g}{l}}$

$\downarrow$

$\frac{d\theta}{dt} = \frac{d\theta}{d\tau} \sqrt{\frac{g}{l}}, \quad \frac{d^2\theta}{dt^2} = \frac{g}{l} \frac{d^2\theta}{d\tau^2}$

$\downarrow$

$\frac{d^2\theta}{d\tau^2} + \frac{b}{m\sqrt{g l}}\frac{d\theta}{d\tau} + \sin(\theta) = \frac{H(t)}{m g}$

$\downarrow$

# Parmetros da simulao adimensional
dtau = 0.01
tempo_max_adim = 50
N = int(tempo_max_adim / dtau)

# Escala temporal real (em segundos)
tempo_max_real = tempo_max_adim * np.sqrt(l/g)
tempo_real = np.linspace(0, tempo_max_real, N)

# Inicializar arrays
Theta = np.zeros(N)
vel = np.zeros(N)
acel = np.zeros(N)

# Condies iniciais
Theta[0] = np.radians(0)  # Comea em 0 graus
vel[0] = 0.0

# Primeiro passo (Euler modificado)
acel[0] = -np.sin(Theta[0]) - B[0]*vel[0] + A[0]
Theta[1] = Theta[0] + vel[0]*dtau + 0.5*acel[0]*dtau**2

    Theta[i+1] = 2*Theta[i] - Theta[i-1] + acel[i]*dtau**2

# Converter para graus
Theta_graus = np.degrees(Theta)
